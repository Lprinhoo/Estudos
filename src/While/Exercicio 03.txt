/*
Um Posto de combustíveis deseja determinar qual de seus produtos tem a preferência de seus clientes. Escreva
um algoritmo para ler o tipo de combustível abastecido (codificado da seguinte forma: 1.Álcool 2.Gasolina 3.Diesel
4.Fim). Caso o usuário informe um código inválido (fora da faixa de 1 a 4) deve ser solicitado um novo código (até
que seja válido). O programa será encerrado quando o código informado for o número 4. Deve ser escrito a
mensagem: "MUITO OBRIGADO" e a quantidade de clientes que abasteceram cada tipo de combustível, conforme
exemplo.
*/

import java.util.Scanner;

void main() {

    Scanner sc = new Scanner(System.in);

    int alcool = 0, gasolina = 0, diesel = 0, i = 0;

    while(i == 0) {

        System.out.println("Informe um código dentro da faixa dos produtos:");
        System.out.println("[1]Alcool  [2]Gasolina  [3]Diesel  [4]Fim");
        System.out.print("Código: ");
        int num = sc.nextInt();

        switch (num) {
            case 1:
                alcool += 1;
                System.out.println("Dado Adicionado!");
                System.out.println("");
                break;
            case 2:
                gasolina += 1;
                System.out.println("Dado Adicionado!");
                System.out.println("");
                break;
            case 3:
                diesel += 1;
                System.out.println("Dado Adicionado!");
                System.out.println("");
                break;
            case 4:
                System.out.println("MUITO OBRIGADO");
                System.out.println(alcool + " Clientes abasteceram com Alcool");
                System.out.println(gasolina + " Clientes abasteceram com Gasolina");
                System.out.println(diesel + " Clientes abasteceram com Diesel");
                i = 1;
                break;
            default:
                System.out.println("Codigo Invalido");
                System.out.println("");
                break;
        }
    }

    sc.close();

}
