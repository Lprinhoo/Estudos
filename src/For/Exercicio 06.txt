// Ler um número inteiro N e calcular todos os seus divisores.

import java.util.Scanner;

void main() {

    Scanner sc = new Scanner(System.in);

    System.out.print("Digite um número: ");
    int n = sc.nextInt();

    for(int i = 1; i <= n; i++) {

        if(n % i == 0) {
            System.out.println(i);
        }
    }
}
