// Ler um número inteiro N e calcular todos os seus divisores.

import java.util.Scanner;

void main() {

    Scanner sc = new Scanner(System.in);

    System.out.print("Digite um número: ");
    int n = sc.nextInt();

    for (int i=1; i<=n; i++) {

        int primeiro = i;
        int segundo  = i * i;
        int terceiro = i * i * i;
        System.out.printf("%d %d %d%n", primeiro, segundo, terceiro);
    }

}
